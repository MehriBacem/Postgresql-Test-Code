	DROP table  IF EXISTS doc_tags_id;
	DROP table  IF EXISTS doc_tags_text;
	DROP table IF EXISTS tags;
	DROP table IF EXISTS doc_tags_json;
	DROP table IF EXISTS doc_tags_array;
	DROP IF EXISTS table documents;

	CREATE table documents( doc_id serial not null primary key, description text not null );

	CREATE table doc_tags_text (
	        doc_id INT not null references documents(doc_id),
	        tag text not null
	    );
	    create unique index doc_tags_text_doc_id_tag on doc_tags_text(doc_id, tag);
	  create  index doc_tags_text_tag on doc_tags_text(tag) ;

	create table tags (
	        tag_id serial not null primary key,
	        tag text not null unique
	    );

	 create   table doc_tags_id (
	        doc_id int not null references documents(doc_id),
	        tag_id int not null references tags(tag_id)
	    );
	 create   unique index doc_tags_id_doc_id_tag_id on doc_tags_id(doc_id, tag_id);
	 create   index doc_tags_id_tag_id on doc_tags_id(tag_id);


	   create  table doc_tags_json (
	        doc_id int not null references documents(doc_id),
	        tags jsonb
	    );
	 create   unique index doc_tags_id_doc_id on doc_tags_json(doc_id);
	 create   index doc_tags_id_tags on doc_tags_json using gin(tags);

	 create   table doc_tags_array (
	        doc_id int not null references documents(doc_id),
	        tags text[] not null default '{}'
	    );
	 create   unique index doc_tags_id_doc_id on doc_tags_array(doc_id);
	 create   index doc_tags_id_tags on doc_tags_array using gin(tags);
